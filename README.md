## 安装依赖

```bash
npm install wei-photo --save
```

#### main.js
```js
import Vue from 'vue'
import App from './App.vue' 

import weiPhoto from 'wei-photo'
import "wei-photo/style/index.css"
Vue.use(weiPhoto);

Vue.config.productionTip = false;
new Vue({
  render: h => h(App),
}).$mount('#app')

```
